import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

# plot a 3d sin wave
x = np.linspace(-np.pi, np.pi, 201)
y = np.linspace(-np.pi, np.pi, 201)
X, Y = np.meshgrid(x, y)
Z = np.sin(X) * np.sin(Y)

# save the figure
plt.figure()
plt.pcolor(X, Y, Z)
plt.savefig('sin_wave.png')
