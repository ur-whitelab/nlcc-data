import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

# plot a 3d sin wave
x = np.linspace(-np.pi, np.pi, 100)
y = np.linspace(-np.pi, np.pi, 100)
X, Y = np.meshgrid(x, y)
Z = np.sin(X) * np.sin(Y)

# plot
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.plot_surface(X, Y, Z)

# save
plt.savefig('test.png')
