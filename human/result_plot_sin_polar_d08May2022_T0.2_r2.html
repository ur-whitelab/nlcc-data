import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

# plot a sin wave in polar coordinates from 0 to 4 pi

# create the data
theta = np.linspace(0, 4*np.pi, 100)
r = np.sin(theta)

# set up the polar plot
ax = plt.subplot(111, projection='polar')
ax.plot(theta, r)

# save the plot
plt.savefig('polar.png')
