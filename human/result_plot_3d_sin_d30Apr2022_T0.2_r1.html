import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

# plot a 3d sin wave
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
x = np.arange(-10, 10, 0.1)
y = np.arange(-10, 10, 0.1)
x, y = np.meshgrid(x, y)
z = np.sin(x) + np.cos(y)
ax.plot_surface(x, y, z)
plt.savefig('test.png')
