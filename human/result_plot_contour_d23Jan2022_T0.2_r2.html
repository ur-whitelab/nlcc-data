import numpy as np
import matplotlib

import matplotlib.pyplot as plt

Add a color bar

def energy(x,y):
    return 0.5*(x**2+np.sin(y*5))

x = np.arange(-3,3,0.1)
y = np.arange(-3,3,0.1)
X, Y = np.meshgrid(x,y)

energy_data = energy(X,Y)

fig = plt.figure()
ax = fig.add_subplot(111)

ax.contourf(X,Y,energy_data)
ax.contour(X,Y,energy_data,colors='k')
