import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

"""
Create matrix of ones
"""
a = np.ones((5,10))
# end

# plot a 3d sin wave
x = np.linspace(-np.pi, np.pi, 201)
y = np.linspace(-np.pi, np.pi, 201)
X, Y = np.meshgrid(x, y)
Z = np.sin(X) * np.sin(Y)

fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.plot_surface(X, Y, Z)
plt.savefig('3d_sin.png')
