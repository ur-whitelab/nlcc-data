import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

"""
Make a scatter plot with random data
Color the points by distance from the origin
Scale the point size by y value
Add a color bar
Make it horizontal
Use the color scheme from red to blue
Make it a square aspect ratio
save to points.png
"""

# Generate random data
x = np.random.rand(100)
y = np.random.rand(100)

# Calculate distance from origin
r = np.sqrt(x**2 + y**2)

# Make a scatter plot
plt.scatter(x, y, s=y*1000, c=r, cmap='RdBu', alpha=0.5)

# Add a color bar
plt.colorbar()

# Make it horizontal
plt.gca().invert_xaxis()

# Make it a square aspect ratio
plt.gca().set_aspect('equal')

# Save to points.png
plt.savefig('points.png')
