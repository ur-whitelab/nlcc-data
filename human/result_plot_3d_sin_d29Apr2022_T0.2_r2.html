import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt

# plot a 3d sin wave
x = np.linspace(-np.pi, np.pi, 201)
y = np.linspace(-np.pi, np.pi, 201)
X, Y = np.meshgrid(x, y)
Z = np.sin(X) * np.sin(Y)

fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')
ax.plot_surface(X, Y, Z, rstride=1, cstride=1, cmap='hot')
ax.set_xlabel('X')
ax.set_ylabel('Y')
ax.set_zlabel('Z')
plt.savefig('3d_sin.png')
